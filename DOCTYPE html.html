<script>

    function calculate() {

        const staff = parseFloat(document.getElementById('staff').value) || 0;

        const exits = parseFloat(document.getElementById('exits').value) || 0;

        const salary = parseFloat(document.getElementById('salary').value) || 0;

        const hiringInput = parseFloat(document.getElementById('hiring').value) || 0;

 

        const rate = staff > 0 ? (exits / staff) * 100 : 0;

        const totalHiring = hiringInput * exits;

        const totalTraining = (salary * 0.5) * exits;

        const totalProd = (salary * 1.5) * exits;   

        const grandTotal = totalHiring + totalTraining + totalProd;

 

        const fmt = (num) => '$' + num.toLocaleString('es-MX', { minimumFractionDigits: 2 });

 

        document.getElementById('turnoverRate').innerText = rate.toFixed(1) + '%';

        document.getElementById('resHiring').innerText = fmt(totalHiring);

        document.getElementById('resTraining').innerText = fmt(totalTraining);

        document.getElementById('resProd').innerText = fmt(totalProd);

        document.getElementById('resTotal').innerText = fmt(grandTotal);

    }

</script>
